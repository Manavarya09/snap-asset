COMMANDS

# Snap a whole URL
snap-asset https://myapp.com --name hero

# Snipe a specific element
snap-asset https://myapp.com --selector ".feature-card" --name feature

# Raid an entire site
snap-asset extract https://competitor.com --out public/

# Isolate a component
snap-asset component ./src/Hero.tsx --name hero

# Bulk barrage
snap-asset batch

# Night mode
snap-asset https://myapp.com --dark --name hero-night
