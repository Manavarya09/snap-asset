Attack Commands!

# Capture a URL
snap-asset https://myapp.com --name hero

# Capture a specific element
snap-asset https://myapp.com --selector ".card" --name feature

# Extract entire website
snap-asset extract https://competitor.com --out public/

# Capture component in isolation
snap-asset component ./src/Hero.tsx --name hero

# Batch mode
snap-asset batch

# Dark mode variant
snap-asset https://myapp.com --dark --name hero-dark
